import { StrictMode } from 'react';
import ReactDOM from 'react-dom/client';
import { QueryClient, QueryClientProvider } from '@tanstack/react-query';

import './styles/fonts.css';
import './styles/style.css';

import App from './App';

const queryClient = new QueryClient({
  defaultOptions: {
    queries: {
      staleTime: 1000 * 60 * 60 * 24, // how long fetched data is considered ‚Äúfresh‚Äù before it becomes ‚Äústale‚Äù ü°¢ 24 hrs
      gcTime: 1000 * 60 * 60 * 24, // how long inactive (unused) query data stays in memory before being deleted ü°¢ 24 hrs
    },
  },
});

const rootElement = document.getElementById('app');
if (rootElement && !rootElement.innerHTML) {
  const root = ReactDOM.createRoot(rootElement);
  root.render(
    <StrictMode>
      <QueryClientProvider client={queryClient}>
        <App />
      </QueryClientProvider>
    </StrictMode>,
  );
}
